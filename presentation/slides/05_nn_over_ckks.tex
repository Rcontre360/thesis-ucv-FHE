\begin{frame}{Aproximación de Funciones de Activación}
    \begin{itemize}
        \item El esquema CKKS solo soporta sumas y multiplicaciones polinomiales.
        \item \textbf{Desafío:} Funciones no lineales (ReLU, Sigmoid) no pueden evaluarse directamente.
        \item \textbf{Soluciones de Aproximación:}
        \begin{itemize}
            \item \textbf{Series de Taylor:} Útiles para funciones suaves (Sigmoid, Tanh).
            \[ \text{Sigmoid}(x) \approx \frac{1}{2} + \frac{1}{4}x - \frac{1}{48}x^3 + \dots \]
            \item \textbf{Polinomios de Bajo Grado:} Reducen la profundidad del circuito.
            \[ \text{ReLU}(x) \approx x^2 \text{ (CryptoNets)} \quad \text{o} \quad ax^2 + bx + c \]
            \item \textbf{Mínimos Cuadrados:} Ajuste óptimo en un intervalo definido.
        \end{itemize}
    \end{itemize}
\end{frame}
