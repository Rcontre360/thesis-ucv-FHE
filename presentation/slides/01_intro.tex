\section{Introducción y Motivación}

\begin{frame}{Introducción y Motivación}
    \begin{itemize}
        \item \textbf{Adopción Masiva:} Las redes neuronales artificiales han transformado industrias clave como la medicina, las finanzas y la ciberseguridad.
        \item \textbf{Procesamiento de Datos Sensibles:} El éxito de estos modelos depende del acceso a grandes volúmenes de información, a menudo de carácter privado o confidencial.
        \item \textbf{El Desafío de la Confianza:} El uso de servicios en la nube para inferencia plantea riesgos de exposición de datos ante terceros no autorizados.
        \item \textbf{Necesidad de Privacidad:} Es imperativo desarrollar soluciones que permitan el cómputo de modelos complejos sin comprometer la integridad ni el secreto de la información del usuario.
    \end{itemize}
\end{frame}

\begin{frame}{Cifrado Homomórfico (HE) como Solución}
    \begin{itemize}
        \item \textbf{Definición:} Computar directamente sobre datos cifrados.
        \item \textbf{Propiedad:} El resultado, una vez descifrado, coincide con el obtenido en texto plano.
        \item \textbf{Beneficio:} El servidor nunca tiene acceso a la información sensible.
    \end{itemize}
