\section{Fundamentos de Redes Neuronales}

\begin{frame}{Neurona Artificial y Capas}
    \begin{itemize}
        \item Estructura básica: Entrada, Pesos, Sesgo y Función de Activación.
        \item Organización en capas: Entrada, Ocultas y Salida.
        \item Operación fundamental: $h = a[\beta + \Omega x]$.
    \end{itemize}
\end{frame}

\begin{frame}{Redes Superficiales vs. Profundas}
    \begin{block}{Red Superficial (Shallow NN)}
        Consiste en una única capa oculta entre la entrada y la salida:
        \[ \hat{y} = \sigma(\mathbf{W}^{(2)} \cdot \sigma(\mathbf{W}^{(1)} \mathbf{x} + \mathbf{b}^{(1)}) + \mathbf{b}^{(2)}) \]
    \end{block}

    \begin{block}{Red Profunda (Deep NN)}
        Composición de múltiples transformaciones lineales y no lineales:
        \[ \mathbf{h}^{(l)} = \sigma(\mathbf{W}^{(l)} \mathbf{h}^{(l-1)} + \mathbf{b}^{(l)}) \]
        donde la salida final es la composición:
        \[ \hat{y} = (\mathbf{f}_L \circ \mathbf{f}_{L-1} \circ \dots \circ \mathbf{f}_1)(\mathbf{x}) \]
    \end{block}

    \begin{itemize}
        \item La profundidad permite la extracción jerárquica de características complejas.
    \end{itemize}
\end{frame}

\begin{frame}{Redes Neuronales Convolucionales (CNN)}
    Especializadas en datos con estructura espacial, utilizan la operación de convolución para extraer características jerárquicas.

    \begin{block}{Operación de Convolución Discreta 2D}
        Para una entrada $\mathbf{x}$ y un núcleo (kernel) $\mathbf{w}$ de tamaño $k \times k$:
        \[ z_{i,j} = \sum_{m=0}^{k-1} \sum_{n=0}^{k-1} w_{m,n} \cdot x_{i+m, j+n} + b \]
    \end{block}

    \begin{itemize}
        \item \textbf{Mapas de Características:} El resultado $z$ captura patrones locales (bordes, texturas).
        \item \textbf{Propiedades:} Invarianza espacial y compartición de pesos (reduce parámetros).
        \item \textbf{Capas:} Combinación de convoluciones, \textit{Pooling} y activaciones.
    \end{itemize}
\end{frame}
