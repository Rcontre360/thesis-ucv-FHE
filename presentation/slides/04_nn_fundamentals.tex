\section{Fundamentos de Redes Neuronales}

\begin{frame}{Neurona Artificial y Capas}
    \begin{itemize}
        \item Estructura básica: Entrada, Pesos, Sesgo y Función de Activación.
        \item Organización en capas: Entrada, Ocultas y Salida.
        \item Operación fundamental: $h = a[\beta + \Omega x]$.
    \end{itemize}
\end{frame}

\begin{frame}{Redes Superficiales vs. Profundas}
    \begin{itemize}
        \item \textbf{Red Superficial (Shallow):} Arquitectura que cuenta con una única capa oculta entre la entrada y la salida.
        \item \textbf{Red Profunda (Deep):} Composición de múltiples capas ocultas (dos o más), lo que permite aprender representaciones jerárquicas y abstractas.
    \end{itemize}

\begin{frame}{Redes Neuronales Convolucionales (CNN)}
    Especializadas en datos con estructura espacial, utilizan la operación de convolución para extraer características jerárquicas.

    \begin{itemize}
        \item \textbf{Propiedades:} Invarianza espacial y compartición de pesos (reduce parámetros).
        \item \textbf{Capas:} Combinación de convoluciones, \textit{Pooling} y activaciones.
    \end{itemize}
